#include<stdio.h>
#include<math.h>
struct n
{
double num[4];
double re[2][4];
};

int main()
{
struct n l;
char s[4] = { ‘+’,’-‘,’*’,’/‘ };
int i = 0;
int z = 0;
for (i = 0; i < 4; i++)
{
scanf(“%lf”, &l.num[i]);
}
l.re[0][0] = l.num[0] + l.num[2];
l.re[1][0] = l.num[1] + l.num[3];
l.re[0][1] = l.num[0] - l.num[2];
l.re[1][1] = l.num[1] - l.num[3];
l.re[0][2] = (l.num[0] * l.num[2]) - (l.num[1] * l.num[3]);
l.re[1][2] = (l.num[0] * l.num[3]) + (l.num[1] * l.num[2]);
l.re[0][3] = ((l.num[0] * l.num[2]) + (l.num[1] * l.num[3])) / (pow(l.num[3], 2) + pow(l.num[2], 2));
l.re[1][3] = ((l.num[1] * l.num[2]) - (l.num[0] * l.num[3])) / (pow(l.num[3], 2) + pow(l.num[2], 2));
i = 0;
while (z < 4)
{
if (l.num[1] < 0 && l.num[3] > 0)
{
printf(“(%.1lf-%.1lfi) %c (%.1lf+%.1lfi) = “, l.num[0], -l.num[1], s[z], l.num[2], l.num[3]);
}
else if (l.num[1] > 0 && l.num[3] < 0)
{
printf(“(%.1lf+%.1lfi) %c (%.1lf-%.1lfi) = “, l.num[0], l.num[1], s[z], l.num[2], -l.num[3]);
}
else if (l.num[1] < 0 && l.num[3] < 0)
{
printf(“(%.1lf-%.1lfi) %c (%.1lf-%.1lfi) = “, l.num[0], -l.num[1], s[z], l.num[2], -l.num[3]);
}
else
{
printf(“(%.1lf+%.1lfi) %c (%.1lf+%.1lfi) = “, l.num[0], l.num[1], s[z], l.num[2], l.num[3]);
}
if (fabs(l.re[0][i]) < 0.1 && fabs(l.re[1][i]) <0.1)
{
printf(“0.0\n”);
}
if (fabs(l.re[0][i]) > 0.1 && fabs(l.re[1][i]) < 0.1)
{
printf(“%.1lf\n”, l.re[0][i]);
}
if (fabs(l.re[0][i]) < 0.1 && fabs(l.re[1][i]) > 0.1)
{
printf(“%.1lfi\n”, l.re[1][i]);
}
if (fabs(l.re[0][i]) > 0.1 && fabs(l.re[1][i]) > 0.1)
{
if (l.re[1][i] > 0)
{
printf(“%.1lf+%.1lfi\n”, l.re[0][i], l.re[1][i]);
}
if (l.re[1][i] < 0)
{
printf(“%.1lf-%.1lfi\n”, l.re[0][i], -l.re[1][i]);
}
}
z++;
i++;
}
return 0;
}
